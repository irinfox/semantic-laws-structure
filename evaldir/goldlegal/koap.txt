Статья 16.12 . 
Нарушение сроков представления таможенной декларации 
Нарушение установленных сроков представления в таможенный орган таможенной декларации , документов или дополнительных сведений , необходимых для таможенных целей , - 
влечет предупреждение или наложение административного штрафа на граждан в размере от одного до трех минимальных размеров оплаты труда ; на должностных лиц - от трех до пяти минимальных размеров оплаты труда ; на юридических лиц - от тридцати до пятидесяти минимальных размеров оплаты труда . 
	LegalInfo
	{
		Subject = нарушение установленных сроков представления в таможенный орган таможенной декларации, документов или дополнительных сведений, необходимых для таможенных целей 
		SubjectKey = нарушение установленных сроков представления
	}
	LegalInfo
	{
		Pred = влечет
		ObjectPlainText = предупреждение или наложение административного штрафа на граждан в размере от одного до трех минимальных размеров оплаты труда
		Penalty = предупреждение или наложение административного штрафа на граждан в размере от одного до трех минимальных размеров оплаты труда
		Inversion = false
	}
	LegalInfo
	{
		Penalty = на должностных лиц - от трех до пяти минимальных размеров оплаты труда 
	}
	LegalInfo
	{
		Penalty = на юридических лиц - от тридцати до пятидесяти минимальных размеров оплаты труда
	}
